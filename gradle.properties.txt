# ============================================
# МАКСИМАЛЬНАЯ ПРОИЗВОДИТЕЛЬНОСТЬ
# ============================================

# Память для Gradle (увеличено с 2GB до 8GB)
org.gradle.jvmargs=-Xmx8g -Xms2g -XX:MaxMetaspaceSize=1g -XX:+HeapDumpOnOutOfMemoryError -XX:+UseParallelGC -Dfile.encoding=UTF-8

# Параллельная сборка (используем все ядра CPU)
org.gradle.parallel=true

# Кэширование сборки (ускоряет повторные сборки в 5-10 раз)
org.gradle.caching=true

# Configuration on demand (собираем только нужное)
org.gradle.configureondemand=true

# Daemon (держит Gradle в памяти между сборками)
org.gradle.daemon=true

# Kotlin incremental compilation (инкрементальная компиляция)
kotlin.incremental=true
kotlin.incremental.usePreciseJavaTracking=true

# Kotlin Daemon (отдельный процесс для компиляции Kotlin)
kotlin.daemon.jvmargs=-Xmx4g -Xms512m -XX:MaxMetaspaceSize=512m

# Kotlin compiler caching
kotlin.compiler.execution.strategy=in-process

# AndroidX
android.useAndroidX=true

# Отключаем лишнее
android.enableJetifier=false
android.enableR8.fullMode=true

# Compose compiler
kotlin.code.style=official
